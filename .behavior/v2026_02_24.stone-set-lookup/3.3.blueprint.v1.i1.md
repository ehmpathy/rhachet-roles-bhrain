## contracts

### `findOneStoneByPattern`

```ts
/**
 * .what = finds exactly one stone matching a pattern
 * .why = enables flexible stone lookup with failfast for ambiguous patterns
 *
 * .note = uses asStoneGlob for pattern expansion (auto-wrap if no glob chars)
 *         throws BadRequestError if multiple stones match
 */
export const findOneStoneByPattern = (input: {
  stones: RouteStone[];
  pattern: string;
}): RouteStone | null;
```

**behavior:**
- calls `asStoneGlob({ pattern })` to expand pattern (e.g., `3.3` ‚Üí `*3.3*`)
- filters stones using `isStoneInGlob({ name, glob })`
- returns single match or `null` if no matches
- throws `BadRequestError` if multiple matches (failfast)

**error format:**
```
pattern "3" matched 4 stones; be more specific
metadata: { pattern: "3", matched: ["3.1.research.v1", "3.2.blueprint.v1", ...] }
```

## implementation details

### step 1: add `findOneStoneByPattern` to `asStoneGlob.ts`

```ts
import { BadRequestError } from 'helpful-errors';

import type { RouteStone } from '@src/domain.objects/Driver/RouteStone';

// ... extant asStoneGlob and isStoneInGlob ...

/**
 * .what = finds exactly one stone matching a pattern
 * .why = enables flexible stone lookup with failfast for ambiguous patterns
 *
 * .note = uses asStoneGlob for pattern expansion (auto-wrap if no glob chars)
 *         throws BadRequestError if multiple stones match
 */
export const findOneStoneByPattern = (input: {
  stones: RouteStone[];
  pattern: string;
}): RouteStone | null => {
  // expand pattern via asStoneGlob (auto-wrap if no glob chars)
  const { glob } = asStoneGlob({ pattern: input.pattern });

  // find all matching stones
  const matched = input.stones.filter((s) =>
    isStoneInGlob({ name: s.name, glob }),
  );

  // failfast if more than one match
  if (matched.length > 1) {
    throw new BadRequestError(
      `pattern "${input.pattern}" matched ${matched.length} stones; be more specific`,
      { pattern: input.pattern, matched: matched.map((s) => s.name) },
    );
  }

  return matched[0] ?? null;
};
```

## test coverage

### unit tests: `asStoneGlob.test.ts`

```ts
import { BadRequestError } from 'helpful-errors';
import { getError, given, then, when } from 'test-fns';

import type { RouteStone } from '@src/domain.objects/Driver/RouteStone';

import { findOneStoneByPattern } from './asStoneGlob';

describe('findOneStoneByPattern', () => {
  const mockStones = [
    { name: '1.vision' },
    { name: '2.criteria' },
    { name: '3.1.research.v1' },
    { name: '3.2.blueprint.v1' },
    { name: '3.3.execute.v1' },
  ] as RouteStone[];

  given('[case1] pattern matches exactly one stone', () => {
    when('[t0] pattern is partial name without glob chars', () => {
      then('returns the matching stone (auto-wrap)', () => {
        const result = findOneStoneByPattern({
          stones: mockStones,
          pattern: 'vision',
        });
        expect(result?.name).toEqual('1.vision');
      });
    });

    when('[t1] pattern is numeric prefix', () => {
      then('returns the matching stone', () => {
        const result = findOneStoneByPattern({
          stones: mockStones,
          pattern: '3.2',
        });
        expect(result?.name).toEqual('3.2.blueprint.v1');
      });
    });

    when('[t2] pattern is exact stone name', () => {
      then('returns the matching stone', () => {
        const result = findOneStoneByPattern({
          stones: mockStones,
          pattern: '3.3.execute.v1',
        });
        expect(result?.name).toEqual('3.3.execute.v1');
      });
    });

    when('[t3] pattern uses explicit glob', () => {
      then('returns the matching stone', () => {
        const result = findOneStoneByPattern({
          stones: mockStones,
          pattern: '*.blueprint.*',
        });
        expect(result?.name).toEqual('3.2.blueprint.v1');
      });
    });
  });

  given('[case2] pattern matches no stones', () => {
    when('[t0] pattern has no matches', () => {
      then('returns null', () => {
        const result = findOneStoneByPattern({
          stones: mockStones,
          pattern: 'nonexistent',
        });
        expect(result).toBeNull();
      });
    });
  });

  given('[case3] pattern matches multiple stones', () => {
    when('[t0] pattern is ambiguous', () => {
      then('throws BadRequestError with matched stone names', async () => {
        const error = await getError(async () =>
          findOneStoneByPattern({
            stones: mockStones,
            pattern: '3.',
          }),
        );
        expect(error).toBeInstanceOf(BadRequestError);
        expect(error.message).toContain('matched 3 stones');
        expect(error.message).toContain('be more specific');
      });
    });

    when('[t1] pattern matches all stones', () => {
      then('throws BadRequestError', async () => {
        const error = await getError(async () =>
          findOneStoneByPattern({
            stones: mockStones,
            pattern: '@all',
          }),
        );
        expect(error).toBeInstanceOf(BadRequestError);
        expect(error.message).toContain('matched 5 stones');
      });
    });
  });
});
```
