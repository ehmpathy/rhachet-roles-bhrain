# 2.1.criteria.blackbox: route guard persistence

## usecase.1 = passage state consolidation

given('a route with stones that can pass or be approved')
  when('a stone passes')
    then('a passage entry is appended to passage.jsonl')
      sothat('all passage state lives in one file')
    then('the entry contains stone name and status')
      sothat('passage can be queried by stone')
    then('no separate .passed file is created')
      sothat('git diff noise is reduced')

  when('a stone is approved')
    then('an approved entry is appended to passage.jsonl')
      sothat('approval state is tracked alongside passage')
    then('no separate .approved file is created')
      sothat('git diff noise is reduced')

  when('a stone is blocked')
    then('a blocker entry is appended to passage.jsonl')
      sothat('blocker state is tracked alongside passage')
    then('no separate .blocker.json file is created')
      sothat('git diff noise is reduced')

given('a route with passage.jsonl')
  when('caller queries if a stone has passed')
    then('the system reads passage.jsonl')
    then('returns true if entry with status=passed found for stone')
    then('returns false if no such entry found')

  when('caller queries if a stone is approved')
    then('returns true if entry with status=approved found for stone')
    then('returns false if no such entry found')


## usecase.2 = gitignore findsert

given('a .route/ directory')
  when('the route is initialized or driven')
    then('.route/.gitignore is findserted')
      sothat('git tracks only .gitignore, passage.jsonl, and .bind.* flags')
    then('the gitignore contains rules to ignore all other files')

given('a .route/ directory with .gitignore already present')
  when('the route is driven again')
    then('the .gitignore is not duplicated')
      sothat('findsert is idempotent')


## usecase.3 = git diff experience

given('a route with guard artifacts and passage.jsonl')
  when('user runs git status')
    then('only .gitignore, passage.jsonl, and .bind.* files appear as tracked')
    then('guard artifacts (*.guard.*) do not appear')
      sothat('git diff shows only what matters')

given('a pr with stone passage')
  when('reviewer views the diff')
    then('they see one line added to passage.jsonl')
    then('they do not see 50+ guard artifact files')
      sothat('pr reviews focus on passage, not cache noise')


## usecase.4 = fresh clone experience

given('a repo cloned fresh')
  when('the clone completes')
    then('.gitignore is present')
    then('passage.jsonl is present with all passage state')
    then('.bind.* flags are present')
    then('guard artifacts are absent')
      sothat('cache is local-only, not versioned')

  when('route.drive runs on fresh clone')
    then('guards re-run because cache is absent')
    then('passage state is read from passage.jsonl')
      sothat('passage is not lost, only cache')


## usecase.5 = concurrent writes

given('two processes write to passage.jsonl simultaneously')
  when('both attempt to append')
    then('writes are serialized via file lock')
      sothat('concurrent drives do not corrupt state')
    then('both entries are written in sequence')
      sothat('no data is lost')
