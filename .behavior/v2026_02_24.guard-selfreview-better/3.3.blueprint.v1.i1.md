1. warm owl/zen frame
2. triggered marker file (use mtime for time enforcement)
3. 30-second time enforcement
4. challenge message when too fast

## domain.operations changes

### setSelfReviewTriggeredReport.ts (NEW)

```typescript
/**
 * .what = write marker file when self-review is first shown
 * .why = file mtime enables time enforcement; content has slug + hash for clarity
 */
export const setSelfReviewTriggeredReport = async (input: {
  stone: string;
  slug: string;
  hash: string;
  route: string;
}): Promise<{ path: string }> => {
  const filename = `${input.stone}.guard.selfreview.${input.slug}.${input.hash}.triggered`;
  const filepath = path.join(input.route, '.route', filename);

  // only write if not exists (preserve original trigger time via mtime)
  if (!await fileExists(filepath)) {
    const content = [
      `slug: ${input.slug}`,
      `hash: ${input.hash}`,
    ].join('\n');
    await fs.writeFile(filepath, content);
  }

  return { path: filepath };
};
```

### getSelfReviewTriggeredReport.ts (NEW)

```typescript
/**
 * .what = get trigger time from marker file mtime
 * .why = enables elapsed time calculation for enforcement
 */
export const getSelfReviewTriggeredReport = async (input: {
  stone: string;
  slug: string;
  hash: string;
  route: string;
}): Promise<{ mtime: Date } | null> => {
  const filename = `${input.stone}.guard.selfreview.${input.slug}.${input.hash}.triggered`;
  const filepath = path.join(input.route, '.route', filename);

  try {
    const stat = await fs.stat(filepath);
    return { mtime: stat.mtime };
  } catch {
    return null;
  }
};
```

### getSelfReviewChallengeDecision.ts (NEW)

```typescript
/**
 * .what = decide if promise should be allowed or challenged
 * .why = encapsulates trigger lookup, time enforcement, and report creation
 */
export const getSelfReviewChallengeDecision = async (input: {
  stone: string;
  slug: string;
  hash: string;
  route: string;
}): Promise<{ decision: 'allowed' } | { decision: 'challenged' }> => {
  const report = await getSelfReviewTriggeredReport(input);

  // no report = challenged, start timer now
  if (!report) {
    await setSelfReviewTriggeredReport(input);
    return { decision: 'challenged' };
  }

  // enforce 30 second minimum
  const elapsed = Date.now() - report.mtime.getTime();
  const threshold = 90 * 1000;

  if (elapsed < threshold) {
    return { decision: 'challenged' };
  }

  return { decision: 'allowed' };
};
```
