# blackbox criteria: fix-achiever

> what experience must be delivered to fulfill the wish

---

# usecase.1 = session lifecycle with goals

given('brain starts a session')
  when('session starts fresh (no compaction)')
    then('briefs about goal usage are booted into context')
      sothat('brain learns how to use goals')
    then('brain sees achiever role is active')

  when('session resumes after compaction')
    then('onBoot hook fires')
    then('goal state is refreshed into context via goal.triage.next output')
      sothat('brain remembers unfinished goals')

given('brain attempts to stop')
  when('unfinished goals remain')
    then('onStop hook fires with reminder')
    then('reminder lists each unfinished goal with status')
      sothat('brain knows what remains')

  when('unfinished goals remain after 5 repeated stop attempts')
    then('onStop hook escalates message intensity')
    then('message grows more insistent with each attempt')
      sothat('brain understands urgency')

  when('all goals are fulfilled')
    then('brain can stop without reminder')

# usecase.2 = goal creation

given('brain receives an ask from human')
  when('brain creates a goal with all required fields')
    then('goal is persisted')
    then('goal appears in goal.triage.next output')
    then('goal appears in goal.triage.infer output')

given('brain creates a goal via flags')
  when('all 6 fields are provided one-by-one')
    then('goal is created with all fields populated')
    then('no error is raised')

  when('any required field is absent')
    then('fail-fast with helpful error')
    then('error explains which field is absent')
    then('error shows example of correct usage')

given('brain creates a goal via stdin yaml')
  when('yaml contains all required fields')
    then('goal is created')

  when('yaml contains unknown keys')
    then('fail-fast with helpful error')
    then('error lists each unknown key')
    then('error shows allowed keys')

# usecase.4 = scope detection

given('brain is bound to a route')
  when('brain invokes goal.memory.set without --scope')
    then('scope is automatically detected as route scope')
    then('goal is persisted under route path')

  when('brain invokes goal.memory.set with --scope repo')
    then('fail-fast with helpful error')
    then('error explains scope is automatic when bound to route')
    then('error suggests to remove --scope flag')

given('brain is not bound to any route')
  when('brain invokes goal.memory.set without --scope')
    then('scope defaults to repo scope')
    then('goal is persisted under repo path')

# usecase.7 = escalation with blockers tracker

given('brain has unfinished goals')
  when('onStop hook fires once')
    then('gentle reminder is shown')
    then('blockers count increments to 1')

  when('onStop hook fires 5 times without progress')
    then('message escalates in urgency')
    then('message emphasizes need to do the work')
    then('blockers count is 5')

  when('brain fulfills a goal after reminders')
    then('blockers count resets for that goal')
    then('next reminder is gentle again')
