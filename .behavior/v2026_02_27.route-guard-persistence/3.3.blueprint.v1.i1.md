# 3.3.blueprint.v1: route guard persistence

## overview

consolidate passage markers (passed, approved, blocked) into `passage.jsonl` and gitignore all guard artifacts. this reduces git diff noise from 50+ files to a single append-only file.

## domain.objects

### PassageReport

```ts
interface PassageReport {
  stone: string;
  status: 'passed' | 'approved' | 'blocked';
  blocker?: 'review.self' | 'review.peer' | 'judge' | 'approval';
  reason?: string;
}

class PassageReport extends DomainLiteral<PassageReport> implements PassageReport {}
```

## key operations

### setPassageReport

```ts
/**
 * .what = appends a passage entry to passage.jsonl
 * .why = consolidates passage state into single file
 * .note = POSIX O_APPEND guarantees atomic writes for small entries
 */
export const setPassageReport = async (input: {
  entry: PassageReport;
  route: string;
}): Promise<{ path: string }> => {
  const routeDir = path.join(input.route, '.route');
  const passagePath = path.join(routeDir, 'passage.jsonl');

  await fs.mkdir(routeDir, { recursive: true });
  await fs.appendFile(passagePath, JSON.stringify(input.entry) + '\n');

  return { path: passagePath };
};
```

### getAllPassageReports

```ts
/**
 * .what = reads all passage reports from passage.jsonl
 * .why = enables passage history lookup
 */
export const getAllPassageReports = async (input: {
  route: string;
}): Promise<PassageReport[]> => {
  const passagePath = path.join(input.route, '.route', 'passage.jsonl');

  const exists = await fs.access(passagePath).then(() => true).catch(() => false);
  if (!exists) return [];

  const content = await fs.readFile(passagePath, 'utf-8');
  return content
    .split('\n')
    .filter(Boolean)
    .map((line) => JSON.parse(line) as PassageReport);
};
```

### getOnePassageReport

```ts
/**
 * .what = reads latest passage report for a stone from passage.jsonl
 * .why = enables current passage state lookup
 */
export const getOnePassageReport = async (input: {
  stone: string;
  status?: 'passed' | 'approved' | 'blocked';
  route: string;
}): Promise<PassageReport | null> => {
  const reports = await getAllPassageReports({ route: input.route });

  const matched = reports
    .filter((r) => r.stone === input.stone)
    .filter((r) => !input.status || r.status === input.status);

  return matched.length > 0 ? matched[matched.length - 1]! : null;
};
```

### findsertRouteGitignore

```ts
/**
 * .what = findserts .gitignore into .route/ directory
 * .why = ensures guard artifacts are gitignored while passage is tracked
 */
export const findsertRouteGitignore = async (input: {
  route: string;
}): Promise<{ path: string; action: 'created' | 'unchanged' }> => {
  const routeDir = path.join(input.route, '.route');
  const gitignorePath = path.join(routeDir, '.gitignore');

  const gitignoreContent = `# ignore everything except passage.jsonl and .bind flags
*
!.gitignore
!passage.jsonl
!.bind.*
`;

  // ensure .route dir exists
  await fs.mkdir(routeDir, { recursive: true });

  // check if gitignore exists with correct content
  const extant = await fs.readFile(gitignorePath, 'utf-8').catch(() => null);
  if (extant === gitignoreContent) {
    return { path: gitignorePath, action: 'unchanged' };
  }

  // write gitignore
  await fs.writeFile(gitignorePath, gitignoreContent);
  return { path: gitignorePath, action: 'created' };
};
```

## migration notes

### backwards compatibility

- extant `.passed`, `.approved`, `.blocker.json` files remain valid
- read operations check both old files and passage.jsonl
- new writes go to passage.jsonl only

### migration strategy

1. deploy code that writes to passage.jsonl
2. read operations fall back to old files if passage.jsonl entry absent
3. optional: migration command to convert old files to passage.jsonl entries
4. remove fallback reads after full migration
