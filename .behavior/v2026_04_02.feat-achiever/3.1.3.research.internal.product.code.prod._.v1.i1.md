### [5] node -e argument parse pattern [REUSE]

**source:** `src/contract/cli/route.ts`

**pattern:**
```typescript
const isNodeEvalMode = (): boolean => {
  return process.argv[1] === undefined || process.argv[1].startsWith('[');
};

const parseArgs = (argv: string[]): Record<string, string> => {
  const args: Record<string, string> = {};
  const startIndex = isNodeEvalMode() ? 2 : 2;
  for (let i = startIndex; i < argv.length; i++) {
    const arg = argv[i];
    if (arg?.startsWith('--') && argv[i + 1]) {
      args[arg.slice(2)] = argv[i + 1]!;
      i++;
    }
  }
  return args;
};
```

**relevance:** achiever cli operations need same argument parse for `--scope`, `--slug`, `--status`, `--covers`.

**disposition:** [REUSE] — use extant `parseArgs` utility

}): Promise<void> => {
  const gitignorePath = path.join(input.route, '.route', '.gitignore');
  const gitignoreContent = '# route state\n*\n';
  try {
    await fs.access(gitignorePath);
  } catch {

};
