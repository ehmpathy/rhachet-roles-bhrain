# blackbox criteria: route malfunction resolve

## usecase.1 = malfunction recovery via passage

given(a stone has malfunction status recorded in passage.jsonl)
  when(the stone is subsequently marked as passed)
    then(passage.jsonl contains both entries in order)
      sothat(the history is preserved for audit)
    then(the latest status for that stone is 'passed')
      sothat(log semantics are honored - last entry wins)
  when(route.drive --mode hook is invoked)
    then(the hook does NOT halt with malfunction error)
      sothat(the mechanic can proceed normally after fix)
    then(exit code is 0 or 2, not 3)
      sothat(malfunction escalation is not triggered

given(a stone has only malfunction status in passage.jsonl)
  when(route.drive --mode hook is invoked)
    then(the hook halts with exit code 3)
      sothat(malfunction is escalated to human
    then(output mentions malfunction)
      sothat(the cause is clear

## usecase.2 = reviewer malfunction cache exclusion

given(a reviewer malfunctions for artifact hash X)
  when(the reviewer is fixed and stone.set is retried for same hash X)
    then(a fresh review runs - not cached)
      sothat(the fixed reviewer gets a chance to run
    then(a new review artifact is created)
      sothat(the fix is verified

given(a reviewer succeeds for artifact hash X)
  when(stone.set is retried for same hash X)
    then(the cached review is reused)
      sothat(redundant compute is avoided
    then(no new review artifact is created)
      sothat(cache integrity is maintained

## usecase.3 = malfunction detection accuracy

given(multiple stones with mixed statuses in passage.jsonl)
  when(one stone has latest status = malfunction)
    then(route.drive --mode hook halts)
      sothat(any unresolved malfunction blocks progress
  when(all stones have latest status != malfunction)
    then(route.drive --mode hook proceeds normally)
      sothat(settled malfunctions don't block forever
