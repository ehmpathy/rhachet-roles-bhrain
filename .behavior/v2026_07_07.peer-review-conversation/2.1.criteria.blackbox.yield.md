# 2.1 blackbox criteria — peer-review conversation

experience boundaries only — what must be delivered, not how. bdd `given/when/then/sothat`.

vocabulary (from vision):
- `.given.by_peer.$slug` — a reviewer's critique (blockers + nitpicks), written by the guard
- `.taken.by_self.$slug` — the driver's contemplation, paired to a `.given` by `(stone, iter, hash, idx, slug)`
- `--as contemplated --that $slug` — the driver's signal that it engaged reviewer `$slug`
- `$conversation` — opt-in glob of a stone's `.given` + `.taken` files, handed to a reviewer

---

# usecase.1 = driver contemplates each reviewer, then the stone progresses

given('a stone gated on two peer reviewers, architect and mechanic')
  given('the guard has run both, so a .given.by_peer exists for each')

    when('the driver writes a .taken.by_self for architect and signals --as contemplated --that architect')
      then('the guard records architect as contemplated')
      then('the guard still blocks progress')
        sothat('a driver cannot skip the second reviewer')

    when('the driver writes a .taken.by_self for mechanic and signals --as contemplated --that mechanic')
      then('every current-iteration .given now has a paired .taken')
      then('the stone is allowed to progress')
        sothat('progress proves the driver engaged every critique')

---

# usecase.2 = driver tries to progress without contemplation — blocked with a legible reply prompt

given('a stone with two peer .given files and zero .taken files')

  when('the driver attempts --as arrived')
    then('the stone is blocked')
    then('the stdout lists each reviewer, its slug, and its verdict (N blockers, N nitpicks)')
    then('under each reviewer it shows, one above the other, contemplate-from (the .given path) and articulate-into (the .taken path)')
      sothat('the driver knows exactly where to read and where to write, per reviewer')

  when('the driver attempts --as passed instead')
    then('the same reply-prompt stdout is shown')
      sothat('no progress verb can bypass the contemplation gate')

  when('the route.drive stophook fires while responses are absent')
    then('the same reply-prompt stdout is shown')
      sothat('the driver cannot quietly stop mid-conversation')

---

# usecase.3 = driver signals --as contemplated before the .taken exists — crystal-clear absent message

given('a peer .given for architect exists but its paired .taken does not')

  when('the driver signals --as contemplated --that architect')
    then('the stone is blocked')
    then('the stdout names the exact absent .taken path it looked for')
    then('the stdout shows the paired .given path to contemplate from')
    then('the stdout states why the file is required (the .taken is the proof of contemplation)')
      sothat('the driver never guesses where to put the file or why')

---

# usecase.4 = partial contemplation — some reviewers answered, not all

given('three peer reviewers with three .given files')
  given('the driver has written .taken files for two of them')

  when('the driver attempts --as passed')
    then('the stone is blocked')
    then('the stdout distinguishes the contemplated reviewers from the one still absent')
      sothat('the driver sees precisely which reviewer remains')

---

# usecase.5 = a clean review requires no contemplation

given('a peer reviewer returns 0 blockers and 0 nitpicks')

  when('the driver attempts to progress with no .taken for that reviewer')
    then('the clean reviewer does not block progress')
      sothat('a driver is not forced into busywork when there is no critique to answer')

---

# usecase.6 = a conversation-enabled reviewer sees the prior exchange

given('a reviewer declared with the opt-in conversation flag')
  given('a prior round produced a .given (a blocker) and the driver produced a paired .taken (a claimed fix)')

  when('the guard runs that reviewer for the next round')
    then('the reviewer receives both the prior .given and the paired .taken as reference context')
    then('the reviewer can drop the prior blocker when the fix is present')
    then('the reviewer can re-raise the prior blocker when the fix is absent')
      sothat('the reviewer converges instead of a blind re-raise of a settled critique')

---

# usecase.7 = a reviewer without the conversation flag stays blind (opt-in)

given('a reviewer declared without the conversation flag')
  given('prior-round .given and .taken files exist for the stone')

  when('the guard runs that reviewer for the next round')
    then('the reviewer receives no conversation context')
    then('the reviewer behaves exactly as it does today')
      sothat('a reviewer that works better blind is unaffected by this wish')

---

# usecase.8 = a new iteration invalidates a stale contemplation

given('a reviewer produced a .given at one content hash, and the driver paired a .taken')
  given('the stone artifact changed, so the guard re-runs the reviewer at a new hash and iteration')

  when('the driver attempts to progress on the strength of the old .taken')
    then('the stone is blocked')
    then('the guard treats the prior .taken as stale and requires a fresh contemplation for the new .given')
      sothat('a driver cannot recycle an old reply to satisfy a fresh critique')

---

# usecase.9 = the conversation is legible as a paired transcript

given('a stone that went through two peer-review rounds')

  when('a human opens the stone review location')
    then('each reviewer critique (.given) sits beside its driver reply (.taken), matched by a shared key')
      sothat('a human reads the review as a back-and-forth dialogue, not a pile of one-sided critiques')

---

# usecase.10 = self-review absent-file message parity

given('a self-review is triggered and its articulation file is absent')
  given('the articulation level is assigned by the guard and may differ from a prior attempt')

  when('the driver signals the self-review promise before the articulation file exists')
    then('the stdout prints the exact current-level articulation path verbatim')
    then('the stdout states that the guard assigns the level and the driver must copy the path as-is')
      sothat('a driver never computes a wrong level and stumbles on where the file belongs')

---

# usecase.11 = the reply-prompt renders identically across all three surfaces

given('a stone with peer .given files and absent .taken files')

  when('the driver reaches the halt via the stophook')
    then('the reply-prompt stdout is shown')
  when('the driver reaches the halt via --as arrived')
    then('the identical reply-prompt stdout is shown')
  when('the driver reaches the halt via --as passed')
    then('the identical reply-prompt stdout is shown')
      sothat('the halt experience is consistent no matter which action the driver took')

---

# usecase.12 = a nitpick-only review requires no contemplation

> threaded back from blueprint design-note B8 (i8-r12-blocker-2) — the fourth quadrant the vision left open. paired with usecase.5 (the clean 0/0 quadrant).

given('a peer reviewer returns 0 blockers and N nitpicks, N > 0')

  when('the driver attempts to progress with no .taken for that reviewer')
    then('the nitpick-only reviewer does not block progress')
      sothat('a driver need not formally answer a non-gate suggestion, as nitpicks never gate elsewhere')

---

# usecase.13 = collided blockers surface in a fixed precedence

given('a stone in two blocked states at once — e.g. one reviewer budget-exhausted AND another reviewer uncontemplated')

  when('the driver attempts to progress')
    then('the higher-precedence blocker is the one surfaced, in the order malfunction > constraint > exhausted > contemplation')
      sothat('the driver is pointed at the most-blocked, least-driver-fixable issue first, never a lower-priority prompt that would mislead')

---

# usecase.14 = a route mid-flight when this ships degrades gracefully

given('a stone that already holds a peer .given written before this feature shipped, with no .taken ever expected')

  when('the driver next attempts to progress')
    then('the stone does not hard-fail or crash on the absent .taken')
    then('the already-written .given is treated as prior-generation, and the standard reply-prompt guides the driver')
      sothat('a rollout never self-inflicts the very budget-waste friction the wish exists to remove')
