# blackbox criteria: route.stone.add

## usecase.1 = add stone from stdin

```
given(driver is bound to a route)
  when(route.stone.add --stone 3.1.6.research.custom --from @stdin --mode apply)
    then(stone file is created at $route/3.1.6.research.custom.stone)
      sothat(driver can add research checkpoints on the fly)
    then(content matches stdin input)
    then(exit code is 0)

given(driver is bound to a route)
  when(route.stone.add --stone 3.1.6.research.custom --from @stdin)
    then(plan mode shows preview of stone to be created)
      sothat(driver can verify before commit)
    then(stone file is NOT created)
    then(exit code is 0)
```

## usecase.2 = add stone from template

```
given(driver is bound to a route)
given(template file exists at $behavior/refs/template.research.adhoc.stone)
  when(route.stone.add --stone 3.1.6.research.adhoc --from template($behavior/refs/template.research.adhoc.stone) --mode apply)
    then(stone file is created at $route/3.1.6.research.adhoc.stone)
    then(content matches template file content)
    then(exit code is 0)

given(driver is bound to a route)
given(template file does NOT exist at path)
  when(route.stone.add --stone 3.1.6.research --from template($behavior/refs/nonexistent.stone) --mode apply)
    then(failfast with error: "template file not found: $path")
    then(exit code is non-zero)
      sothat(driver knows the template path is wrong)
```

## usecase.4 = explicit route override

```
given(driver is NOT bound to any route)
given(a route exists at path .behavior/v2026_04_12.myroute)
  when(route.stone.add --stone 3.1.6.research --from 'content' --route .behavior/v2026_04_12.myroute --mode apply)
    then(stone file is created at .behavior/v2026_04_12.myroute/3.1.6.research.stone)
    then(exit code is 0)
      sothat(driver can add stones to any route explicitly)
```

## edgecase.1 = no bound route

```
given(driver is NOT bound to any route)
given(--route is NOT provided)
  when(route.stone.add --stone 3.1.6.research --from 'content' --mode apply)
    then(failfast with error: "no route bound; use `rhx route.bind.set` first")
    then(exit code is non-zero)
      sothat(driver knows they need to bind first)
```

## edgecase.2 = stone already exists

```
given(driver is bound to a route)
given(stone 3.1.6.research.custom already exists)
  when(route.stone.add --stone 3.1.6.research.custom --from 'content' --mode apply)
    then(failfast with error: "stone already exists; use different name or `route.stone.del` first")
    then(stone file is NOT modified)
    then(exit code is non-zero)
      sothat(driver cannot accidentally overwrite extant stones)
```

## edgecase.3 = invalid stone name

```
given(driver is bound to a route)
  when(route.stone.add --stone 'research' --from 'content' --mode apply)
    then(failfast with error: "stone name must have numeric prefix followed by at least one alpha segment")
    then(exit code is non-zero)
      sothat(stone names remain consistent)

given(driver is bound to a route)
  when(route.stone.add --stone '123' --from 'content' --mode apply)
    then(failfast with error: "stone name must have numeric prefix followed by at least one alpha segment")
    then(exit code is non-zero)

given(driver is bound to a route)
  when(route.stone.add --stone '3.1.6.research.custom' --from 'content' --mode apply)
    then(succeeds — valid stone name)
    then(exit code is 0)
```

## edgecase.4 = empty stdin

```
given(driver is bound to a route)
given(stdin is empty)
  when(route.stone.add --stone 3.1.6.research --from @stdin --mode apply)
    then(failfast with error: "no content provided via stdin")
    then(exit code is non-zero)
      sothat(driver knows content is required)
```

## edgecase.5 = absent --from

```
given(driver is bound to a route)
  when(route.stone.add --stone 3.1.6.research --mode apply)
    then(failfast with error that --from is required)
    then(exit code is non-zero)
      sothat(driver knows content source is required)
```

## edgecase.6 = $behavior variable expansion

```
given(driver is bound to a route at .behavior/v2026_04_12.myroute)
given(template exists at .behavior/v2026_04_12.myroute/refs/template.stone)
  when(route.stone.add --stone 3.1.6.research --from template($behavior/refs/template.stone) --mode apply)
    then($behavior expands to .behavior/v2026_04_12.myroute)
    then(template is read from .behavior/v2026_04_12.myroute/refs/template.stone)
    then(exit code is 0)
      sothat(templates are portable across routes)
```
