## usecase.1 = research and document replic brain architecture

given('a learner who wants to understand how replic brains work')
  when('they access the brain architect briefs')
    then('they find a clear definition of atomic vs replic brain frames')
      sothat('they understand the fundamental distinction before diving deeper')
    then('they find cited sources for each architectural claim')
      sothat('they can verify and explore further on their own')
    then('they find cited academic papers where applicable')
      sothat('they have rigorous foundations for the concepts')

given('a learner exploring the replic brain architecture')
  when('they read the atomic concepts breakdown')
    then('each concept is defined independently')
      sothat('they can understand one piece without needing the whole')
    then('each concept has a clear boundary of responsibility')
      sothat('they know exactly what it does and does not cover')
    then('concepts are ordered from foundational to composite')
      sothat('they can build understanding incrementally')

given('a learner who has understood the atomic concepts')
  when('they view the treestruct of how concepts compose')
    then('they see which concepts depend on which')
      sothat('they understand the layering of the architecture')
    then('they see the dataflow between components')
      sothat('they understand how information moves through the system')
    then('they see the control flow between components')
      sothat('they understand how decisions propagate')

## usecase.2 = create and maintain brain architect role

given('a brain architect role does not exist')
  when('the role is created')
    then('it exists at src/roles/architect/')
      sothat('it follows the repo structure conventions')
    then('it has a readme.md declaring its purpose')
      sothat('newcomers understand what this role does')
    then('it has briefs structured as campus course materials')
      sothat('it matches the pattern of src/roles/thinker/briefs/')

given('the brain architect role exists')
  when('someone wants to learn brain architecture')
    then('they find articles explaining core concepts')
      sothat('they can read and understand theory')
    then('they find demos showing concepts in action')
      sothat('they can see practical applications')
    then('they find catalogs of architectural patterns')
      sothat('they can reference and reuse proven approaches')

given('new research emerges on brain architectures')
  when('the brain architect reviews the research')
    then('relevant findings are added to the briefs')
      sothat('the role stays current with the field')
    then('sources are cited with dates')
      sothat('readers know the recency of information')
    then('outdated information is marked or removed')
      sothat('readers are not misled by stale content')

## usecase.3 = document reference blueprint for claude-code replic brain

given('a learner who wants to understand how claude-code is built')
  when('they access the reference blueprint')
    then('they find a structural diagram of the full replic brain')
      sothat('they can see the whole system at a glance')
    then('they find the exact sequence of operations in a replic loop')
      sothat('they understand the runtime behavior')
    then('they find the boundaries between llm invocation and tool execution')
      sothat('they understand where the "atomic" meets the "replic"')

given('the reference blueprint is documented')
  when('a reader examines the components')
    then('each component is described at pseudocode level')
      sothat('they understand the logic without implementation details')
    then('each component shows inputs and outputs')
      sothat('they understand the data contracts')
    then('each component explains its responsibility')
      sothat('they understand why it exists')

given('the reference blueprint is documented')
  when('a reader examines the repl loop')
    then('they see the read phase (user input, context assembly)')
      sothat('they understand how the brain receives information')
    then('they see the execute phase (llm invocation, tool calls)')
      sothat('they understand how the brain acts')
    then('they see the print phase (response formatting, state update)')
      sothat('they understand how the brain responds')
    then('they see the loop continuation logic')
      sothat('they understand when the brain stops or continues')

given('the reference blueprint is educational')
  when('compared to actual claude-code implementation')
    then('it captures the essential architecture accurately')
      sothat('readers gain true understanding')
    then('it omits implementation-specific optimizations')
      sothat('readers focus on concepts not details')
    then('it notes where simplifications were made')
      sothat('readers know what was abstracted away')
