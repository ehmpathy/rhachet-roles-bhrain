## test sketches

### repro.1: clone runs `--as arrived`

```typescript
given('[case1] clone marks stone as arrived', () => {
  const scene = useBeforeAll(async () => {
    const tempDir = genTempDir({
      slug: 'arrived-alias',
      clone: path.join(ASSETS_DIR, 'route.simple'),
    });
    return { tempDir };
  });

  when('[t1] clone runs --as arrived', () => {
    const result = useThen('stone is evaluated', async () =>
      stepRouteStoneSet(
        { stone: '1.vision', route: scene.tempDir, as: 'arrived' },
        noopContext,
      ),
    );

    then('stone is marked as passed', () => {
      expect(result.passed).toBe(true);
    });
  });
});
```

### repro.2: normal guidance (hooks < 7)

```typescript
given('[case2] route.drive with hooks < 7', () => {
  const scene = useBeforeAll(async () => {
    const tempDir = genTempDir({
      slug: 'drive-normal',
      clone: path.join(ASSETS_DIR, 'route.simple'),
    });
    // set hook counter to 3
    await setDriveHookState({ route: tempDir, stone: '1.vision', hooksWithoutAttempt: 3 });
    return { tempDir };
  });

  when('[t1] route.drive fires', () => {
    const result = useThen('drive output generated', async () =>
      stepRouteDrive({ route: scene.tempDir, mode: 'hook' }, noopContext),
    );

    then('output shows --as arrived command', () => {
      expect(result.emit?.stdout).toContain('--as arrived');
    });

    then('output shows stone content', () => {
      expect(result.emit?.stdout).toContain("here's the stone");
    });

    then('output does NOT show drum nudge', () => {
      expect(result.emit?.stdout).not.toContain('walk the way');
    });
  });
});
```


### repro.4: hook counter resets on attempt

```typescript
given('[case4] hook counter resets on passage attempt', () => {
  const scene = useBeforeAll(async () => {
    const tempDir = genTempDir({
      slug: 'counter-reset',
      clone: path.join(ASSETS_DIR, 'route.simple'),
    });
    // set hook counter to 10
    await setDriveHookState({ route: tempDir, stone: '1.vision', hooksWithoutAttempt: 10 });
    return { tempDir };
  });

  when('[t1] clone runs --as arrived', () => {
    useThen('stone set executes', async () =>
      stepRouteStoneSet(
        { stone: '1.vision', route: scene.tempDir, as: 'arrived' },
        noopContext,
      ),
    );

    then('hook counter is reset to 0', async () => {
      const state = await getDriveHookState({ route: scene.tempDir, stone: '1.vision' });
      expect(state.hooksWithoutAttempt).toBe(0);
    });
  });
});
```
