# usecase.1 = ask accumulation via onTalk

```
given('human sends a message to claude code')
  when('the UserPromptSubmit hook fires')
    then('the ask is appended to asks.inventory.jsonl')
      sothat('no ask is forgotten between messages')
    then('a reminder is emitted to stderr')
      sothat('the brain is prompted to consider goal impact')
    then('the hook exits 0')
      sothat('the brain continues without halt')

given('human sends multiple messages in a session')
  when('each UserPromptSubmit hook fires')
    then('each ask is appended in order')
      sothat('the audit trail preserves message sequence')
    then('duplicate messages create separate entries')
      sothat('re-emphasis is captured')

given('human sends an empty message')
  when('the UserPromptSubmit hook fires')
    then('no ask is appended')
      sothat('the inventory is not polluted with noise')
    then('the hook exits 0 silently')
      sothat('the brain is not distracted')
```

# usecase.2 = coverage verification via onStop

```
given('a session with accumulated asks')
  when('the session ends and onStop fires')
    then('uncovered asks are enumerated')
      sothat('the brain sees what needs triage')
    then('the brain is halted until triage complete')
      sothat('no ask escapes without a goal')

given('all asks are covered by goals')
  when('the session ends and onStop fires')
    then('the brain is not halted')
      sothat('covered sessions end smoothly')
```
