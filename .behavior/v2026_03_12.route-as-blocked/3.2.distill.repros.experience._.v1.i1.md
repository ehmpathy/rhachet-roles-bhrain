### test sketch: first block attempt

```typescript
given('[case5] set stone as blocked without articulation', () => {
  const tempDir = path.join(os.tmpdir(), `test-step-set-blocked-${Date.now()}`);

  beforeEach(async () => {
    await fs.cp(path.join(ASSETS_DIR, 'route.simple'), tempDir, { recursive: true });
  });

  afterEach(async () => {
    await fs.rm(tempDir, { recursive: true, force: true });
  });

  when('[t0] --as blocked without articulation', () => {
    then('shows nudge and creates triggered file', async () => {
      const result = await stepRouteStoneSet(
        { stone: '1.vision', route: tempDir, as: 'blocked' },
        noopContext,
      );
      expect(result.challenged).toBe(true);
      expect(result.emit?.stdout).toContain('failure is only the opportunity to begin again');
      expect(result.emit?.stdout).toContain('passage = unchanged');

      // verify triggered file created
      const routeDir = path.join(tempDir, '.route');
      const files = await fs.readdir(routeDir);
      const triggeredFile = files.find(f => f.includes('.blocked.triggered'));
      expect(triggeredFile).toBeDefined();
    });
  });
});
```

### test sketch: successful block

```typescript
given('[case6] set stone as blocked with articulation', () => {
  const tempDir = path.join(os.tmpdir(), `test-step-set-blocked-success-${Date.now()}`);

  beforeEach(async () => {
    await fs.cp(path.join(ASSETS_DIR, 'route.simple'), tempDir, { recursive: true });
    // create triggered file (simulate prior attempt)
    await fs.mkdir(path.join(tempDir, '.route'), { recursive: true });
    await fs.writeFile(
      path.join(tempDir, '.route', '1.vision.blocked.triggered'),
      new Date().toISOString(),
    );
    // create articulation file
    await fs.mkdir(path.join(tempDir, '.route', 'blocker'), { recursive: true });
    await fs.writeFile(
      path.join(tempDir, '.route', 'blocker', '1.vision.md'),
      '# blocker\n\n## what blocks me\ntest\n\n## what i tried\ntest\n\n## what i need\ntest\n',
    );
  });

  afterEach(async () => {
    await fs.rm(tempDir, { recursive: true, force: true });
  });

  when('[t0] --as blocked with articulation', () => {
    then('records blocked status', async () => {
      const result = await stepRouteStoneSet(
        { stone: '1.vision', route: tempDir, as: 'blocked' },
        noopContext,
      );
      expect(result.blocked).toBe(true);
      expect(result.emit?.stdout).toContain('passage = blocked');

      // verify passage.jsonl updated
      const passagePath = path.join(tempDir, '.route', 'passage.jsonl');
      const content = await fs.readFile(passagePath, 'utf-8');
      expect(content).toContain('"status":"blocked"');
    });
  });
});
```
