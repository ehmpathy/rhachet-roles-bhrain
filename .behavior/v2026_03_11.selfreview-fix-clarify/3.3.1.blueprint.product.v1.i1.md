### formatWalkTheWay.ts [+] create

```typescript
export const formatWalkTheWay = (input: {
  route: string;
  stone: string;
  slug: string;
}): string[] => {
  const articulationPath = `${input.route}/review/self/${input.stone}.${input.slug}.md`;
  const lines: string[] = [];

  // [+] walk the way header
  lines.push(`   в”ңв”Җ walk the way рҹӘ·`);
  lines.push(`   в”Ӯ  в”Ӯ`);

  // [+] articulate into (nested)
  lines.push(`   в”Ӯ  в”ңв”Җ articulate into`);
  lines.push(`   в”Ӯ  в”Ӯ  в””в”Җ ${articulationPath}`);
  lines.push(`   в”Ӯ  в”Ӯ`);

  // [+] for each found issue drum block
  lines.push(`   в”Ӯ  в”ңв”Җ for each found issue рҹӘҳ`);
  lines.push(`   в”Ӯ  в”Ӯ  в”ңв”Җ`);
  lines.push(`   в”Ӯ  в”Ӯ  в”Ӯ`);
  lines.push(`   в”Ӯ  в”Ӯ  в”Ӯ  articulate how it was fixed`);
  lines.push(`   в”Ӯ  в”Ӯ  в”Ӯ  so you remember for next time`);
  lines.push(`   в”Ӯ  в”Ӯ  в”Ӯ`);
  lines.push(`   в”Ӯ  в”Ӯ  в””в”Җ`);
  lines.push(`   в”Ӯ  в”Ӯ`);

  // [+] for each non issue drum block
  lines.push(`   в”Ӯ  в””в”Җ for each non issue рҹӘҳ`);
  lines.push(`   в”Ӯ     в”ңв”Җ`);
  lines.push(`   в”Ӯ     в”Ӯ`);
  lines.push(`   в”Ӯ     в”Ӯ  articulate why it holds`);
  lines.push(`   в”Ӯ     в”Ӯ  so others can learn from it`);
  lines.push(`   в”Ӯ     в”Ӯ`);
  lines.push(`   в”Ӯ     в””в”Җ`);

  return lines;
};
```
